## 📌코드

------

```c++
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
vector<int> graph[501], reverseGraph[501];
bool visited[501];
int N, M, a, b, answer, cnt;

void input(){
    cin >> N >> M;
    for(int i = 0; i < M; i++){
        cin >> a >> b;
        graph[a].push_back(b);
        reverseGraph[b].push_back(a);
    }
}

void init(){
    fill(&visited[0], &visited[501], false);
    cnt = 0;
}

void dfs(int idx){
    visited[idx] = true;
    for(auto it : graph[idx]){
        if(visited[it]){
            continue;
        }
        cnt++;
        dfs(it);
    }
}

void reverseDfs(int idx){
    visited[idx] = true;
    for(auto it : reverseGraph[idx]){
        if(visited[it]){
            continue;
        }
        cnt++;
        reverseDfs(it);
    }
}

void solve(){
    input();
    for(int i = 1; i <= N; i++){
        int temp = 0;

        init();
        dfs(i);
        temp += cnt;

        init();
        reverseDfs(i);
        temp += cnt;
        
        if(temp == N - 1){
            answer++;
        }
    }
    cout << answer;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```
