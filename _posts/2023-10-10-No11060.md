## :pushpin:풀이과정



나는 Bottom-up 방식의 Dp로 풀이했다.

각 단계에서 점프할 수 있는 범위를 계산하여 최소화 해주었다.



## :pushpin:코드



```c++
#include <bits/stdc++.h>
#define INF 100000000
using namespace std;
int N, jump[1001], dp[1001];

void input(){
    cin >> N;
    for(int i = 1; i <= N; i++){
        cin >> jump[i];
    }
}

void init(){
    for(int i = 1; i <= N; i++){
        dp[i] = INF;
    }
}

void solve(){
    init();
    dp[1] = 0;
    for(int i = 1; i <= N; i++){
        for(int j = 1; j <= jump[i]; j++){
            if(i + j > N){
                continue;
            }
            dp[i + j] = min(dp[i + j], dp[i] + 1);
        }
    }
    if(dp[N] == INF){
        cout << -1;
    }
    else{
        cout << dp[N];
    }
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    input();
    solve();
    return 0;
}
```
