## :pushpin:풀이과정





동전 문제와 비슷하다. 각 단계의 숫자를 전 단계의 숫자에서 더하거나 빼는게 가능하면 경우의 수들을 더한다.







## :pushpin:코드





```c++
#include <iostream>
using namespace std;
int N, arr[101];
long long dp[101][21];

void input(){
    cin >> N;
    for(int i = 0; i < N; i++){
        cin >> arr[i];
    }
}

void solve(){
    dp[0][arr[0]] = 1;
    for(int i = 1; i < N - 1; i++){
        for(int j = 0; j <= 20; j++){
            if(!dp[i - 1][j]){
                continue;
            }
            if(j + arr[i] <= 20){
                dp[i][j + arr[i]] += dp[i - 1][j];
            }
            if(j - arr[i] >= 0){
                dp[i][j - arr[i]] += dp[i - 1][j];
            }
        }
    }
    cout << dp[N - 2][arr[N - 1]] << "\n";
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    input();
    solve();
    return 0;
}
```
