## :pushpin:코드





```c++
#include <iostream>
#include <vector>
using namespace std;
vector<int> arr;
string A, B;

void input(){
    cin >> A >> B;
}

void solve(){
    for(int i = 0; i < A.length(); i++){
        arr.push_back(A[i] - '0');
        arr.push_back(B[i] - '0');
    }
    while(1){
        vector<int> temp;
        for(int i = 0; i < arr.size() - 1; i++){
            temp.push_back((arr[i] + arr[i + 1]) % 10);
        }
        arr = temp;
        if(arr.size() <= 2){
            cout << arr[0] << arr[1] << "\n";
            break;
        }
    }
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    input();
    solve();
    return 0;
}
```
