## :pushpin:풀이과정





1 더하기와 2 곱하기의 비용이 같으므로 무조건 2곱하기가 유리하다





## :pushpin:코드



```c++
#include <iostream>
#include <algorithm>
#define INF 23412341
using namespace std;
int A, K, dp[1000001];


void input(){
    cin >> A >> K;
}

void init(){
    for(int i = 0; i <= K; i++){
        dp[i] = INF;
    }
}

void solve(){
    init();
    dp[A] = 0;
    for(int i = A + 1; i <= K; i++){
        if(i % 2 == 0){
            dp[i] = dp[i / 2] + 1;
        }
        dp[i] = min(dp[i], dp[i - 1] + 1);
    }
    cout << dp[K] << "\n";
}

int main(){
    input();
    solve();
    return 0;
}
```
