## 📌문제 설명  

------
기본 그래프 탐색 문제이다. 1부터 출발하여 N에 도착하면 cnt를 출력해주면 된다.

## 📌코드

------

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;
bool visited[10001];
vector<int> graph[10001];
int T, N, A;


void init(){
    for(int i = 1; i <= N; i++){
        visited[i] = false;
    }
    for(int i = 1; i <= N; i++){
        graph[i].clear();
    }
}

void bfs(){
    queue<pair<int, int>> q;
    q.push({1, 0});
    visited[1] = true;
    while(!q.empty()){
        auto [cur, cnt] = q.front();
        q.pop();
        if(cur == N){
            cout << cnt << '\n';
            return;
        }
        for(auto it : graph[cur]){
            if(!visited[it]){
                visited[it] = true;
                q.push({it, cnt + 1});
            }
        }
    }
    cout << 0 << '\n';
}

void input(){
    cin >> T;
    while(T--){
        cin >> N;
        init();
        for(int i = 1; i <= N; i++){
            cin >> A;
            graph[i].push_back(A);
        }
        bfs();
    }
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    input();
    return 0;
}
```
