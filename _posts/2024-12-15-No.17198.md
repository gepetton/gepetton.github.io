## 📌문제 설명  

------
B부터 시작해서 L까지의 최단거리를 찾아주면 된다.

## 📌코드

------

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;
bool visited[11][11];
char mp[11][11];
int dx[4]= {0, 0, 1, -1};
int dy[4]= {1, -1, 0, 0};
int startx, starty, endx, endy;
struct comp{
    int x, y, cnt;
};


void input(){
    for(int i = 0; i < 10; i++){
        for(int j = 0; j < 10; j++){
            cin >> mp[i][j];
            if(mp[i][j] == 'B'){
                startx = i;
                starty = j;
            }
            else if(mp[i][j] == 'L'){
                endx = i;
                endy = j;
            }
        }
    }
}

void bfs(){
    queue<comp> q;
    q.push({startx, starty, 0});
    visited[startx][starty] = true;
    while(!q.empty()){
        auto [x, y, cnt] = q.front();
        q.pop();
        for(int i = 0; i < 4; i++){
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(nx < 0 || nx >= 10 || ny < 0 || ny >= 10 || visited[nx][ny] || mp[nx][ny] == 'R') continue;
            if(nx == endx && ny == endy){
                cout << cnt;
                return;
            }
            visited[nx][ny] = true;
            q.push({nx, ny, cnt + 1});
        }
    }
}

void solve(){
    input();
    bfs();
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    solve();
    return 0;
}
```
