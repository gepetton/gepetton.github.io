## :pushpin:풀이과정







다른 분들의 풀이를 보니 트라이를 사용한 분들이 많았다. 나는 입력 크기를 보고 unordered_set으로 충분히 가능할 것 같다 판단하고 unordered_set으로 문제를 해결했다.







## :pushpin:코드





```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <unordered_set>
using namespace std;
vector<string> arr;
string str;
int t, n;

void init(){
    arr.clear();
}

void solve(){
    sort(arr.begin(), arr.end());
    unordered_set<string> us;
    for(auto it : arr){
        string temp = "";
        for(int i = 0; i < it.length() - 1; i++){
            temp.push_back(it[i]);
            if(us.find(temp) != us.end()){
                cout << "NO\n";
                return;
            }
        }
        us.insert(it);
    }
    cout << "YES\n";
}

void input(){
    cin >> t;
    while(t--){
        init();
        cin >> n;
        for(int i = 0; i < n; i++){
            cin >> str;
            arr.push_back(str);
        }
        solve();
    }
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    input();
    return 0;
}
```
